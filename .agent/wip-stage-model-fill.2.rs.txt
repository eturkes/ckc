//! SPEC §7.4/§9 model-fill processing stage core — the nondeterministic stage
//! that fills a route's target artifact (ClinicalIR JSON, SMT text, …) from
//! recorded model calls, generic over the target.
//!
//! [`model_fill`] obtains each model output through the [`CassetteStore`] —
//! replay (default, runtime-absent) reads a committed cassette, record (gated)
//! invokes the [`ModelAdapter`] once and writes one — then runs the route's
//! acceptance closure over the decoded bytes. Acceptance is the §4 check, so the
//! target choice (ClinicalIR / SMT) and its acceptance (JSON-Schema + structural
//! / solver parse, plus the route's grounding) stay with the route, which
//! reports each rejection as a [`FillReject`].
//!
//! Two rejection shapes drive the stage:
//! - [`FillReject::Schema`] — output failed schema/parse acceptance → a §7.4
//!   `ai_schema_violation`, and the stage RE-PROMPTS: a repair under a fresh
//!   derived seed, up to `repair_limit` times. Still failing on the last
//!   permitted attempt → a terminal `repair_limit_exceeded`.
//! - [`FillReject::Grounding`] — output parsed but references upstream ids absent
//!   from the deterministic upstream → a §7.4 `ai_hallucinated_source`, terminal
//!   (a hallucination is not a schema defect, so it does not consume repairs).
//!
//! Each attempt is one recorded model call keyed by its own seed (attempt 0 = the
//! base seed, repair `i` = `derive_seed(base, i)`), so a repair sequence replays
//! as distinct committed cassettes. The stage accounts both
//! [`recorded_calls`](ModelFill::recorded_calls) and [`repairs`](ModelFill::repairs)
//! for the §7.3 metrics; the route/run wiring (run-m2.1) builds the §4.6 event
//! from them ([`RECORDED_CALLS_COUNTER`]/[`REPAIRS_COUNTER`]). A cassette
//! IO/contract failure stays a [`CassetteError`], distinct from a model output
//! that fails acceptance.

use std::path::Path;

use ckc_core::{DiagnosticCode, DiagnosticRecord, Id, Outcome};

use crate::cassette::{CassetteError, CassetteKey, CassetteStore, RecordContext};
use crate::model::{ModelAdapter, derive_seed};
use crate::shell::static_id;

/// §4.6 event resource-counter key for the §7.3 recorded-call count: every model
/// call a fill made (the initial draw plus each repair).
pub const RECORDED_CALLS_COUNTER: &str = "recorded_calls";

/// §4.6 event resource-counter key for the §7.3 repair count: re-prompts after a
/// schema violation. Equals `recorded_calls - 1` in this single-draw-per-attempt
/// stage, but is a distinct §7.3 metric — kept separate so run-m2.1 emits both
/// counters and no downstream layer re-derives one from the other.
pub const REPAIRS_COUNTER: &str = "repairs";

/// How [`model_fill`] obtains each attempt's model output. Copied per attempt so
/// the repair loop re-acquires under a fresh derived seed.
#[derive(Clone, Copy)]
pub enum FillSource<'a> {
    /// Replay the committed cassette — never touches the runtime (default).
    Replay,
    /// Invoke the runtime once per attempt and record it (gated; requires the
    /// adapter, prompt, constraint path, and recording context). A repair
    /// re-sends the same prompt under a fresh derived seed — error-feedback
    /// re-prompting (varying the prompt with the violation) is a route/run
    /// concern, deferred; the seed is what distinguishes the attempt here.
    Record {
        /// Live model-runtime adapter.
        adapter: &'a ModelAdapter,
        /// Prompt sent to the runtime over stdin.
        prompt: &'a str,
        /// Constraint (schema/grammar) path passed to the runtime.
        constraint: &'a Path,
        /// Provenance + budget for the recording.
        ctx: &'a RecordContext,
    },
}

/// The route's §4 acceptance verdict when it rejects a model output. The variant
/// selects the §7.4 diagnostic code AND whether the stage repairs.
#[derive(Debug)]
pub enum FillReject {
    /// Output failed schema/parse acceptance — a repairable defect. Becomes an
    /// `ai_schema_violation` (the reason in the payload); the stage re-prompts.
    Schema(String),
    /// Output parsed but references upstream ids absent from the deterministic
    /// upstream — a terminal hallucination. Becomes an `ai_hallucinated_source`
    /// (the absent ids in the payload); the stage does not repair.
    Grounding(Vec<Id>),
}

/// One model-fill result: the accepted target, or the diagnostics that rejected
/// every attempt, plus exact call/repair accounting.
#[derive(Debug)]
pub struct ModelFill<T> {
    /// The accepted target, or `None` when no attempt passed acceptance (the
    /// diagnostics then hold the schema violations and the terminal
    /// `repair_limit_exceeded` / `ai_hallucinated_source`).
    pub target: Option<T>,
    /// Stage diagnostics in attempt order: one `ai_schema_violation` per
    /// schema-failed attempt, then a terminal `repair_limit_exceeded` (budget
    /// exhausted) or `ai_hallucinated_source` (grounding failed). A
    /// repaired-then-recovered fill keeps its earlier `ai_schema_violation`s
    /// alongside a present `target`.
    pub diagnostics: Vec<DiagnosticRecord>,
    /// Total model calls recorded for this fill — the §7.3 recorded-call count,
    /// carried on the stage's §4.6 event under [`RECORDED_CALLS_COUNTER`]. One
    /// per attempt.
    pub recorded_calls: u64,
    /// Re-prompts after a schema violation — the §7.3 repair count, carried under
    /// [`REPAIRS_COUNTER`]. Zero when the first attempt is accepted or
    /// hallucinated; equals `repair_limit` on exhaustion.
    pub repairs: u64,
}

/// Run one model-fill with repair: obtain each attempt's cassette (replay/record)
/// through `store`, decode it, and run the route's `accept` over the bytes.
///
/// `accept` is the route's §4 acceptance check (generic over the target, so the
/// route selects ClinicalIR / SMT by supplying it). [`FillReject::Schema`]
/// re-prompts up to `repair_limit` times under fresh derived seeds, then yields a
/// terminal `repair_limit_exceeded`; [`FillReject::Grounding`] is terminal at
/// once. A cassette IO/contract failure returns `Err` — the recording is missing
/// or malformed, distinct from an output that fails acceptance.
pub fn model_fill<T>(
    store: &CassetteStore,
    key: &CassetteKey,
    source: FillSource<'_>,
    repair_limit: u32,
    accept: impl Fn(&[u8]) -> Result<T, FillReject>,
) -> Result<ModelFill<T>, CassetteError> {
    let mut diagnostics = Vec::new();
    let mut attempt: u32 = 0;
    loop {
        // Attempt 0 reads the base key directly; each repair re-prompts under a
        // fresh derived seed, so its cassette is a distinct committed file.
        let seed = if attempt == 0 {
            key.seed
        } else {
            derive_seed(key.seed, attempt)
        };
        let attempt_key = CassetteKey {
            route: key.route.clone(),
            source: key.source.clone(),
            seed,
        };
        let cassette = match source {
            FillSource::Replay => store.replay(&attempt_key)?,
            FillSource::Record {
                adapter,
                prompt,
                constraint,
                ctx,
            } => store.record(adapter, &attempt_key, prompt, constraint, ctx)?,
        };
        // The store validated `output_hex` decodability on load, so this holds.
        let output = cassette
            .payload
            .output_bytes()
            .map_err(|_| CassetteError::InvalidHex)?;
        let recorded_calls = u64::from(attempt) + 1;
        let repairs = u64::from(attempt);
        match accept(&output) {
            Ok(target) => {
                return Ok(ModelFill {
                    target: Some(target),
                    diagnostics,
                    recorded_calls,
                    repairs,
                });
            }
            Err(FillReject::Grounding(absent)) => {
                diagnostics.push(ai_hallucinated_source(absent));
                return Ok(ModelFill {
                    target: None,
                    diagnostics,
                    recorded_calls,
                    repairs,
                });
            }
            Err(FillReject::Schema(reason)) => {
                diagnostics.push(ai_schema_violation(reason));
                if attempt == repair_limit {
                    diagnostics.push(repair_limit_exceeded(repair_limit));
                    return Ok(ModelFill {
                        target: None,
                        diagnostics,
                        recorded_calls,
                        repairs,
                    });
                }
                attempt += 1;
            }
        }
    }
}

/// The §7.4 `ai_schema_violation` diagnostic: model output failed the stage's §4
/// acceptance check (parse or schema). Outcome `invalid`, mirroring the
/// deterministic `schema_invalid`; the reason rides the structured payload.
fn ai_schema_violation(reason: String) -> DiagnosticRecord {
    DiagnosticRecord {
        code: DiagnosticCode::AiSchemaViolation,
        outcome: Outcome::Invalid,
        payload: vec![(static_id("reason"), reason)],
        region_ids: Vec::new(),
        artifact_hashes: Vec::new(),
    }
}

/// The §7.4 `ai_hallucinated_source` diagnostic: a schema-valid output references
/// upstream ids absent from the deterministic upstream. Outcome `invalid`; the
/// sorted absent ids ride the payload. They stay OUT of `region_ids` — §4.5
/// requires those to resolve to real source spans, and a hallucinated id resolves
/// to none.
fn ai_hallucinated_source(mut absent: Vec<Id>) -> DiagnosticRecord {
    absent.sort();
    absent.dedup();
    let joined = absent
        .iter()
        .map(|id| id.to_string())
        .collect::<Vec<_>>()
        .join(" ");
    DiagnosticRecord {
        code: DiagnosticCode::AiHallucinatedSource,
        outcome: Outcome::Invalid,
        payload: vec![(static_id("absent_source_ids"), joined)],
        region_ids: Vec::new(),
        artifact_hashes: Vec::new(),
    }
}

/// The §7.4 `repair_limit_exceeded` diagnostic: the output still failed schema
/// acceptance after the repair budget was exhausted. Outcome `invalid`; the
/// exhausted limit rides the payload as a canonical decimal.
fn repair_limit_exceeded(repair_limit: u32) -> DiagnosticRecord {
    DiagnosticRecord {
        code: DiagnosticCode::RepairLimitExceeded,
        outcome: Outcome::Invalid,
        payload: vec![(static_id("repair_limit"), repair_limit.to_string())],
        region_ids: Vec::new(),
        artifact_hashes: Vec::new(),
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use crate::model::derive_seed;
    use ckc_core::{CassettePayload, ModelIdentity, Producer, hash_bytes};
    use std::collections::BTreeSet;

    fn key_at(seed: u64) -> CassetteKey {
        CassetteKey {
            route: "route.fixture".parse().unwrap(),
            source: "test_source.fixture".parse().unwrap(),
            seed,
        }
    }

    fn key() -> CassetteKey {
        key_at(42)
    }

    fn producer() -> Producer {
        Producer {
            pipeline_id: static_id("pipe.test"),
            pipeline_step_id: static_id("processing_stage.test.model_fill"),
            toolchain_manifest_hash: hash_bytes(b"toolchain"),
        }
    }

    fn temp_store(tag: &str) -> (CassetteStore, std::path::PathBuf) {
        let dir = std::env::temp_dir().join(format!("ckc-model-fill-{tag}-{}", std::process::id()));
        let _ = std::fs::remove_dir_all(&dir);
        (CassetteStore::new(&dir), dir)
    }

    // Persist a synthetic cassette carrying `output` for runtime-absent replay,
    // using the store's own contract-valid wrapper builder.
    fn seed_cassette(store: &CassetteStore, k: &CassetteKey, output: &[u8]) {
        let payload = CassettePayload::from_output(
            k.route.clone(),
            k.source.clone(),
            k.seed,
            "prompt body".to_owned(),
            hash_bytes(b"constraint"),
            hash_bytes(b"prompt template"),
            ModelIdentity {
                model_id: "model.fixture".parse().unwrap(),
                quant: "fixture_quant".to_owned(),
                runtime_version: "1.0.0".to_owned(),
            },
            output,
        );
        let wrapper = store.build_wrapper(k, payload, producer()).unwrap();
        store.persist(k, wrapper).unwrap();
    }

    // The deterministic upstream id-set a route grounds references against.
    fn upstream() -> BTreeSet<Id> {
        ["seg.1", "seg.2"]
            .iter()
            .map(|s| s.parse().unwrap())
            .collect()
    }

    // A route acceptance stand-in over raw model bytes: the JSON object is the
    // target; the `ok` field is the schema check, and a `ref` field (when
    // present) must name an id in `allowed` or it is a hallucinated source.
    // Mirrors a route's parse-then-ground acceptance.
    fn accept_json(
        allowed: &BTreeSet<Id>,
    ) -> impl Fn(&[u8]) -> Result<serde_json::Value, FillReject> + '_ {
        move |bytes| {
            let value: serde_json::Value = serde_json::from_slice(bytes)
                .map_err(|e| FillReject::Schema(format!("parse: {e}")))?;
            if value.get("ok").is_none() {
                return Err(FillReject::Schema("missing required field `ok`".to_owned()));
            }
            if let Some(reference) = value.get("ref").and_then(|r| r.as_str()) {
                let id: Id = reference
                    .parse()
                    .map_err(|_| FillReject::Schema(format!("bad ref id: {reference}")))?;
                if !allowed.contains(&id) {
                    return Err(FillReject::Grounding(vec![id]));
                }
            }
            Ok(value)
        }
    }

    // Assert a diagnostic is an `ai_schema_violation` of the documented shape
    // (Outcome::Invalid, empty region_ids/artifact_hashes, reason under `reason`)
    // and return its reason.
    fn assert_schema_violation(d: &DiagnosticRecord) -> &str {
        assert_eq!(d.code, DiagnosticCode::AiSchemaViolation);
        assert_eq!(d.outcome, Outcome::Invalid);
        assert!(d.region_ids.is_empty());
        assert!(d.artifact_hashes.is_empty());
        assert_eq!(d.payload.len(), 1);
        assert_eq!(d.payload[0].0, static_id("reason"));
        d.payload[0].1.as_str()
    }

    // A first-attempt accepted fill yields the target with no repairs or
    // diagnostics, and one recorded call.
    #[test]
    fn valid_fill_yields_target_and_no_repairs() {
        let (store, dir) = temp_store("valid");
        let k = key();
        seed_cassette(&store, &k, br#"{"ok":true,"ref":"seg.1"}"#);
        let fill = model_fill(&store, &k, FillSource::Replay, 2, accept_json(&upstream())).unwrap();
        assert!(fill.target.is_some());
        assert!(fill.diagnostics.is_empty());
        assert_eq!(fill.recorded_calls, 1);
        assert_eq!(fill.repairs, 0);
        std::fs::remove_dir_all(&dir).unwrap();
    }

    // Repair-then-recover: attempt 0 schema-fails, the repair at the derived seed
    // is accepted — one repair, two recorded calls, the early violation retained
    // alongside the present target.
    #[test]
    fn schema_violation_then_repair_recovers() {
        let (store, dir) = temp_store("recover");
        let k0 = key();
        seed_cassette(&store, &k0, b"not json");
        seed_cassette(&store, &key_at(derive_seed(42, 1)), br#"{"ok":true}"#);
        let fill =
            model_fill(&store, &k0, FillSource::Replay, 2, accept_json(&upstream())).unwrap();
        assert!(fill.target.is_some());
        assert_eq!(fill.diagnostics.len(), 1);
        assert!(assert_schema_violation(&fill.diagnostics[0]).starts_with("parse:"));
        assert_eq!(fill.recorded_calls, 2);
        assert_eq!(fill.repairs, 1);
        std::fs::remove_dir_all(&dir).unwrap();
    }

    // Exhaustion: every attempt through the budget schema-fails — one violation
    // per attempt then a terminal `repair_limit_exceeded` naming the limit, no
    // target, repairs == limit.
    #[test]
    fn repair_limit_exceeded_after_exhausting_budget() {
        let (store, dir) = temp_store("exhausted");
        let k0 = key();
        seed_cassette(&store, &k0, br#"{"other":1}"#);
        seed_cassette(&store, &key_at(derive_seed(42, 1)), br#"{"other":1}"#);
        seed_cassette(&store, &key_at(derive_seed(42, 2)), br#"{"other":1}"#);
        let fill =
            model_fill(&store, &k0, FillSource::Replay, 2, accept_json(&upstream())).unwrap();
        assert!(fill.target.is_none());
        assert_eq!(fill.recorded_calls, 3);
        assert_eq!(fill.repairs, 2);
        assert_eq!(fill.diagnostics.len(), 4);
        for d in &fill.diagnostics[..3] {
            assert_eq!(assert_schema_violation(d), "missing required field `ok`");
        }
        let last = &fill.diagnostics[3];
        assert_eq!(last.code, DiagnosticCode::RepairLimitExceeded);
        assert_eq!(last.outcome, Outcome::Invalid);
        assert!(last.region_ids.is_empty());
        assert!(last.artifact_hashes.is_empty());
        assert_eq!(
            last.payload,
            vec![(static_id("repair_limit"), "2".to_owned())]
        );
        std::fs::remove_dir_all(&dir).unwrap();
    }

    // A zero repair budget exhausts on the first schema failure — one violation
    // then the terminal exhaustion, zero repairs.
    #[test]
    fn zero_repair_budget_exhausts_on_first_violation() {
        let (store, dir) = temp_store("zerobudget");
        let k = key();
        seed_cassette(&store, &k, b"not json");
        let fill = model_fill(&store, &k, FillSource::Replay, 0, accept_json(&upstream())).unwrap();
        assert!(fill.target.is_none());
        assert_eq!(fill.recorded_calls, 1);
        assert_eq!(fill.repairs, 0);
        assert_eq!(fill.diagnostics.len(), 2);
        assert!(assert_schema_violation(&fill.diagnostics[0]).starts_with("parse:"));
        assert_eq!(
            fill.diagnostics[1].code,
            DiagnosticCode::RepairLimitExceeded
        );
        std::fs::remove_dir_all(&dir).unwrap();
    }

    // Grounding is terminal: a schema-valid output referencing an absent upstream
    // id is an `ai_hallucinated_source` that does NOT consume the repair budget —
    // one call, zero repairs, the absent id in the payload, no target.
    #[test]
    fn hallucinated_source_is_terminal_despite_budget() {
        let (store, dir) = temp_store("hallucinated");
        let k = key();
        seed_cassette(&store, &k, br#"{"ok":true,"ref":"seg.99"}"#);
        let fill = model_fill(&store, &k, FillSource::Replay, 2, accept_json(&upstream())).unwrap();
        assert!(fill.target.is_none());
        assert_eq!(fill.recorded_calls, 1);
        assert_eq!(fill.repairs, 0);
        assert_eq!(fill.diagnostics.len(), 1);
        let d = &fill.diagnostics[0];
        assert_eq!(d.code, DiagnosticCode::AiHallucinatedSource);
        assert_eq!(d.outcome, Outcome::Invalid);
        assert!(d.region_ids.is_empty());
        assert!(d.artifact_hashes.is_empty());
        assert_eq!(
            d.payload,
            vec![(static_id("absent_source_ids"), "seg.99".to_owned())]
        );
        std::fs::remove_dir_all(&dir).unwrap();
    }

    // A missing recording is a store error, not a schema violation — the two
    // failure kinds stay distinct (the route can tell a broken cassette from a
    // bad model output).
    #[test]
    fn missing_cassette_is_error_not_violation() {
        let (store, dir) = temp_store("missing");
        let err = model_fill(
            &store,
            &key(),
            FillSource::Replay,
            2,
            accept_json(&upstream()),
        )
        .unwrap_err();
        assert!(matches!(err, CassetteError::Io(_)));
        let _ = std::fs::remove_dir_all(&dir);
    }
}
