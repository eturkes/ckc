fn direct_smt_accept() -> impl Fn(&[u8]) -> Result<String, FillReject> {   // mirror single_ir_accept shape 607-642 (but SHALLOW, no grounding)
    |bytes| {
        let text = std::str::from_utf8(bytes).map_err(|e| FillReject::Schema(format!("not utf-8: {e}")))?;
        if !text.trim_start().starts_with("(set-logic") { return Err(FillReject::Schema("expected a (set-logic ...) head".to_owned())); }
        if !text.contains("(check-sat)") { return Err(FillReject::Schema("expected a (check-sat) command".to_owned())); }
        Ok(text.to_owned())
    }
}   // NO FillReject::Grounding — the direct route has no source linkage; the solver is the syntactic authority (target_syntax_failure at verify, .5).
