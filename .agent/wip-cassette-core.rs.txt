//! SPEC §4.4/§9 model cassette payload — a recorded model call (prompt + output)
//! as a committed test source artifact, enabling deterministic, runtime-absent
//! replay (`origin = ai_generated`, `evidence_status = evidence_discovery_only`).
//!
//! This is the pure §4.4 payload only (every wrapper payload lives in `ckc-core`);
//! the record/replay IO + model-adapter orchestration live in `ckc-cli`'s `cassette`
//! module. The model output is hex-encoded into canonical JSON so any bytes round-trip
//! losslessly — the recorded bytes ARE the determinism: greedy decode is byte-stable
//! on a fixed runtime but not across environments (§9), so replay reads the committed
//! cassette rather than re-invoking the runtime.

use crate::canon::{
    CanonError, CanonRead, CanonReadError, Canonical, ObjectEmitter, ObjectReader, RawText, Reader,
    emit_u64, read_u64,
};
use crate::id::{Hash, Id};
use crate::plans::ModelIdentity;

/// One recorded model call, keyed by (route, source, seed).
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct CassettePayload {
    /// Route id whose prompt template + constraint produced this call.
    pub route: Id,
    /// Test source id the prompt was grounded on.
    pub source: Id,
    /// Decoding seed (a §9 derived seed) for this sample.
    pub seed: u64,
    /// Exact prompt sent to the model runtime over stdin.
    pub prompt: String,
    /// Raw-byte hash (§4.4 `_hash` rule) of the constraint file fed to `--constraint`.
    pub constraint_hash: Hash,
    /// Raw-byte hash of the route prompt template the prompt was rendered from.
    pub prompt_template_hash: Hash,
    /// Identity of the runtime that produced the output.
    pub model_identity: ModelIdentity,
    /// Lowercase-hex of the model's raw stdout bytes — the recorded output.
    pub output_hex: String,
}

impl CassettePayload {
    /// Build a payload from raw output bytes, hex-encoding the output.
    #[allow(clippy::too_many_arguments)]
    pub fn from_output(
        route: Id,
        source: Id,
        seed: u64,
        prompt: String,
        constraint_hash: Hash,
        prompt_template_hash: Hash,
        model_identity: ModelIdentity,
        output: &[u8],
    ) -> Self {
        Self {
            route,
            source,
            seed,
            prompt,
            constraint_hash,
            prompt_template_hash,
            model_identity,
            output_hex: encode_hex(output),
        }
    }

    /// Decode the recorded output back to its exact bytes.
    pub fn output_bytes(&self) -> Result<Vec<u8>, InvalidCassetteHex> {
        decode_hex(&self.output_hex)
    }
}

impl Canonical for CassettePayload {
    fn emit_canonical(&self, out: &mut Vec<u8>) -> Result<(), CanonError> {
        // Members in ascending key byte order (the reader is positional, §4.3).
        let mut obj = ObjectEmitter::new();
        obj.member("constraint_hash", |b| self.constraint_hash.emit_canonical(b))?;
        obj.member("model_identity", |b| self.model_identity.emit_canonical(b))?;
        obj.member("output_hex", |b| {
            RawText(self.output_hex.clone()).emit_canonical(b)
        })?;
        obj.member("prompt", |b| RawText(self.prompt.clone()).emit_canonical(b))?;
        obj.member("prompt_template_hash", |b| {
            self.prompt_template_hash.emit_canonical(b)
        })?;
        obj.member("route", |b| self.route.emit_canonical(b))?;
        obj.member("seed", |b| {
            emit_u64(b, self.seed);
            Ok(())
        })?;
        obj.member("source", |b| self.source.emit_canonical(b))?;
        obj.finish(out)
    }
}

impl CanonRead for CassettePayload {
    fn read(r: &mut Reader<'_>) -> Result<Self, CanonReadError> {
        let mut obj = ObjectReader::open(r)?;
        let constraint_hash = obj.member("constraint_hash", Hash::read)?;
        let model_identity = obj.member("model_identity", ModelIdentity::read)?;
        let output_hex = obj.member("output_hex", |r| Ok(RawText::read(r)?.0))?;
        let prompt = obj.member("prompt", |r| Ok(RawText::read(r)?.0))?;
        let prompt_template_hash = obj.member("prompt_template_hash", Hash::read)?;
        let route = obj.member("route", Id::read)?;
        let seed = obj.member("seed", read_u64)?;
        let source = obj.member("source", Id::read)?;
        obj.close()?;
        Ok(Self {
            route,
            source,
            seed,
            prompt,
            constraint_hash,
            prompt_template_hash,
            model_identity,
            output_hex,
        })
    }
}

/// Returned by [`CassettePayload::output_bytes`] when `output_hex` is not valid
/// lowercase hex (even length, digits `0-9a-f`).
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct InvalidCassetteHex;

impl std::fmt::Display for InvalidCassetteHex {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        f.write_str("cassette output_hex is not valid lowercase hex")
    }
}

impl std::error::Error for InvalidCassetteHex {}

fn encode_hex(bytes: &[u8]) -> String {
    const HEX: &[u8; 16] = b"0123456789abcdef";
    let mut s = String::with_capacity(bytes.len() * 2);
    for &b in bytes {
        s.push(HEX[(b >> 4) as usize] as char);
        s.push(HEX[(b & 0x0f) as usize] as char);
    }
    s
}

fn decode_hex(s: &str) -> Result<Vec<u8>, InvalidCassetteHex> {
    let bytes = s.as_bytes();
    if !bytes.len().is_multiple_of(2) {
        return Err(InvalidCassetteHex);
    }
    let mut out = Vec::with_capacity(bytes.len() / 2);
    for pair in bytes.chunks_exact(2) {
        let hi = hex_val(pair[0]).ok_or(InvalidCassetteHex)?;
        let lo = hex_val(pair[1]).ok_or(InvalidCassetteHex)?;
        out.push((hi << 4) | lo);
    }
    Ok(out)
}

fn hex_val(c: u8) -> Option<u8> {
    match c {
        b'0'..=b'9' => Some(c - b'0'),
        b'a'..=b'f' => Some(c - b'a' + 10),
        _ => None,
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use crate::canon::read_strict_canonical;
    use crate::hash::hash_bytes;

    fn sample(output: &[u8]) -> CassettePayload {
        CassettePayload::from_output(
            "route.fixture".parse().unwrap(),
            "test_source.fixture".parse().unwrap(),
            42,
            "prompt body".to_owned(),
            hash_bytes(b"constraint"),
            hash_bytes(b"prompt template"),
            ModelIdentity {
                model_id: "model.fixture".parse().unwrap(),
                quant: "fixture_quant".to_owned(),
                runtime_version: "1.0.0".to_owned(),
            },
            output,
        )
    }

    #[test]
    fn canonical_round_trip() {
        // Output carries non-UTF-8 bytes — hex must survive them losslessly.
        let payload = sample(&[0x00, 0xff, 0x10, 0x7b, 0x0a]);
        let mut bytes = Vec::new();
        payload.emit_canonical(&mut bytes).unwrap();
        let back: CassettePayload = read_strict_canonical(&bytes).unwrap();
        assert_eq!(payload, back);
    }

    #[test]
    fn output_bytes_round_trip() {
        let raw = [0x00u8, 0xff, 0x10, 0x7b, 0x0a, 0x80];
        let payload = sample(&raw);
        assert_eq!(payload.output_hex, "00ff107b0a80");
        assert_eq!(payload.output_bytes().unwrap(), raw);
    }

    #[test]
    fn output_bytes_rejects_malformed_hex() {
        let mut payload = sample(b"");
        payload.output_hex = "abc".to_owned(); // odd length
        assert_eq!(payload.output_bytes(), Err(InvalidCassetteHex));
        payload.output_hex = "zz".to_owned(); // non-hex
        assert_eq!(payload.output_bytes(), Err(InvalidCassetteHex));
        payload.output_hex = "AB".to_owned(); // uppercase rejected (lowercase canon)
        assert_eq!(payload.output_bytes(), Err(InvalidCassetteHex));
    }
}
